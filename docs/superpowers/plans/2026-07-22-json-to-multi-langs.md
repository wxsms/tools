## 关键实现细节（所有任务共享）

### quicktype EnumOption 的传值方式

quicktype 的 `EnumOption` 在 `rendererOptions` 里接收的是 **enum key 字符串**，不是 boolean。源码确认：

- Java `array-type`: `new EnumOption("array-type", ..., { array: false, list: true }, "list")` — key 是 `'array'` 或 `'list'`，默认 `'list'`
- C# `array-type`: 同 enum，默认 `'array'`（与 Java 不同！）
- C# `framework`: `{ NewtonSoft: "NewtonSoft", SystemTextJson: "SystemTextJson" }`，默认 `'SystemTextJson'`

所以「数组用 `List<T>` 而非 `T[]`」checkbox 的映射是：
- 勾选 → `rendererOptions['array-type'] = 'list'`
- 不勾选 → `rendererOptions['array-type'] = 'array'`

### `allPropertiesOptional` 是顶层 option

不是 renderer option，跟 Go/TS 一致，传给 `quicktype({...})` 顶层：

```js
const result = await quicktype({
  lang: '...',
  inputData,
  rendererOptions,
  allPropertiesOptional: true,  // 顶层
})
```

### 标识符正则

Python/Rust/Java/C++/C# 标识符基本都可用 `^[A-Za-z_][A-Za-z0-9_]*$`。warning 文案按语言替换「Go」为对应语言名。

### 返回值契约（所有语言一致）

```ts
{ ok: true, code: string, warning?: string }
| { ok: false, error: string }
```

- 空输入 → `{ ok: true, code: '' }`
- JSON 解析失败 → `{ ok: false, error: 'JSON 解析失败：' + e.message }`
- 非法 typeName → `{ ok: true, code, warning: '...不是合法 X 标识符' }`，code 使用 fallback `AutoGenerated`

```js
import { quicktype, jsonInputForTargetLanguage, InputData } from 'quicktype-core'

/**
 * @param {string} jsonString
 * @param {{ allOptional?: boolean, pydantic?: boolean, typeName?: string }} [options]
 * @returns {Promise<{ ok: true, code: string, warning?: string } | { ok: false, error: string }>}
 *
 * quicktype-core Python renderer 行为：
 * - `just-types: true`  只输出 class 声明，不输出 runtime JSON 解析代码
 * - `pydantic-base-model: true`  让类继承 pydantic.BaseModel
 * - 顶层 `allPropertiesOptional: true`  让单样本字段渲染为 `Optional[T]`
 *
 * `typeName` 作为 `addSource({ name })` 传入，控制顶层类名。非合法 Python 标识符时
 * 回退到 'AutoGenerated' 并返回 warning，让 UI 提示用户。
 */
const PY_IDENT_RE = /^[A-Za-z_][A-Za-z0-9_]*$/

export async function jsonToPython(jsonString, {
  allOptional = false,
  pydantic = false,
  typeName = 'AutoGenerated',
} = {}) {
  if (!jsonString || jsonString.trim() === '') {
    return { ok: true, code: '' }
  }

  let parsed
  try {
    parsed = JSON.parse(jsonString)
  } catch (e) {
    return { ok: false, error: 'JSON 解析失败：' + e.message }
  }

  const trimmedName = (typeName || '').trim()
  const isValidName = trimmedName === '' || PY_IDENT_RE.test(trimmedName)
  const safeName = isValidName ? (trimmedName || 'AutoGenerated') : 'AutoGenerated'
  const warning = isValidName
    ? undefined
    : `结构体名 "${trimmedName}" 不是合法 Python 标识符`

  try {
    const jsonInput = jsonInputForTargetLanguage('python')
    await jsonInput.addSource({ name: safeName, samples: [JSON.stringify(parsed)] })
    const inputData = new InputData()
    inputData.addInput(jsonInput)

    const rendererOptions = { 'just-types': true }
    if (pydantic) rendererOptions['pydantic-base-model'] = true
    const extraOptions = {}
    if (allOptional) extraOptions.allPropertiesOptional = true

    const result = await quicktype({
      lang: 'python',
      inputData,
      rendererOptions,
      ...extraOptions,
    })
    return { ok: true, code: result.lines.join('\n'), warning }
  } catch (e) {
    return { ok: false, error: '生成失败：' + e.message }
  }
}
```

```vue
<script setup>
import { ref, onMounted, onBeforeUnmount, watch, nextTick } from 'vue'
import { EditorView, keymap, lineNumbers, highlightActiveLineGutter, highlightActiveLine, drawSelection, rectangularSelection, highlightSpecialChars } from '@codemirror/view'
import { EditorState } from '@codemirror/state'
import { json as jsonLang } from '@codemirror/lang-json'
import { python } from '@codemirror/lang-python'
import { oneDark } from '@codemirror/theme-one-dark'
import { defaultKeymap, history, historyKeymap, indentWithTab } from '@codemirror/commands'
import { syntaxHighlighting, defaultHighlightStyle, bracketMatching } from '@codemirror/language'
import { closeBrackets, closeBracketsKeymap } from '@codemirror/autocomplete'
import { searchKeymap, highlightSelectionMatches } from '@codemirror/search'
import {
  ArrowDownIcon,
  ClipboardDocumentIcon,
  CheckIcon,
  TrashIcon,
} from '@heroicons/vue/24/outline'
import { jsonToPython } from './json-to-python.js'
import { useTheme } from '../../composables/useTheme.js'

const DEFAULT_INPUT = JSON.stringify({
  name: 'wxsm',
  age: 18,
  active: true,
  tags: ['dev', 'python'],
  address: { city: 'BJ', zip: '100000' },
}, null, 2)

const input = ref(DEFAULT_INPUT)
const output = ref('')
const error = ref('')
const warning = ref('')
const allOptional = ref(false)
const pydantic = ref(false)
const typeName = ref('')
const inputCopied = ref(false)
const outputCopied = ref(false)

const inputEditorEl = ref(null)
const outputEditorEl = ref(null)
let inputEditor = null
let outputEditor = null
let debounceTimer = null

const { theme } = useTheme()

function getThemeExt() {
  return theme.value === 'dark' ? oneDark : []
}

function createExtensions(langExt, onChange) {
  return [
    lineNumbers(),
    highlightActiveLineGutter(),
    highlightSpecialChars(),
    history(),
    drawSelection(),
    EditorState.allowMultipleSelections.of(true),
    syntaxHighlighting(defaultHighlightStyle, { fallback: true }),
    bracketMatching(),
    closeBrackets(),
    rectangularSelection(),
    highlightActiveLine(),
    highlightSelectionMatches(),
    keymap.of([
      ...closeBracketsKeymap,
      ...defaultKeymap,
      ...searchKeymap,
      ...historyKeymap,
      indentWithTab,
    ]),
    langExt,
    EditorView.updateListener.of(onChange),
    EditorView.theme({
      '&': { height: '100%' },
      '.cm-scroller': { overflow: 'auto' },
    }),
  ]
}

function inputOnChange(update) {
  if (!update.docChanged) return
  input.value = update.state.doc.toString()
  scheduleGenerate()
}

function outputOnChange() {
  // 输出是只读的，不处理变更
}

function createEditors() {
  if (inputEditorEl.value && !inputEditor) {
    inputEditor = new EditorView({
      state: EditorState.create({
        doc: input.value,
        extensions: [...createExtensions(jsonLang(), inputOnChange), getThemeExt()],
      }),
      parent: inputEditorEl.value,
    })
  }
  if (outputEditorEl.value && !outputEditor) {
    outputEditor = new EditorView({
      state: EditorState.create({
        doc: output.value,
        extensions: [
          ...createExtensions(python(), outputOnChange),
          EditorState.readOnly.of(true),
          getThemeExt(),
        ],
      }),
      parent: outputEditorEl.value,
    })
  }
}

function destroyEditors() {
  inputEditor?.destroy()
  inputEditor = null
  outputEditor?.destroy()
  outputEditor = null
}

async function generate() {
  const r = await jsonToPython(input.value, {
    allOptional: allOptional.value,
    pydantic: pydantic.value,
    typeName: typeName.value.trim() || 'AutoGenerated',
  })
  if (r.ok) {
    error.value = ''
    warning.value = r.warning || ''
    output.value = r.code
    syncOutputEditor()
  } else {
    output.value = ''
    warning.value = ''
    error.value = r.error
    syncOutputEditor()
  }
}

function syncOutputEditor() {
  if (!outputEditor) return
  outputEditor.dispatch({
    changes: { from: 0, to: outputEditor.state.doc.length, insert: output.value },
  })
}

function scheduleGenerate() {
  error.value = ''
  warning.value = ''
  if (debounceTimer) clearTimeout(debounceTimer)
  debounceTimer = setTimeout(generate, 300)
}

function regenerate() {
  if (debounceTimer) {
    clearTimeout(debounceTimer)
    debounceTimer = null
  }
  generate()
}

function clear() {
  input.value = ''
  output.value = ''
  error.value = ''
  warning.value = ''
  if (inputEditor) {
    inputEditor.dispatch({
      changes: { from: 0, to: inputEditor.state.doc.length, insert: '' },
    })
  }
  syncOutputEditor()
}

async function copyText(text, flag) {
  try {
    await navigator.clipboard.writeText(text)
    if (flag === 'inputCopied') {
      inputCopied.value = true
      setTimeout(() => { inputCopied.value = false }, 1500)
    } else {
      outputCopied.value = true
      setTimeout(() => { outputCopied.value = false }, 1500)
    }
  } catch { /* clipboard not available */ }
}

watch(theme, async () => {
  destroyEditors()
  await nextTick()
  createEditors()
  syncOutputEditor()
})

onMounted(() => {
  createEditors()
  generate()
})

onBeforeUnmount(() => {
  if (debounceTimer) clearTimeout(debounceTimer)
  destroyEditors()
})
</script>
```

```js
import { quicktype, jsonInputForTargetLanguage, InputData } from 'quicktype-core'

/**
 * @param {string} jsonString
 * @param {{ deriveDebug?: boolean, deriveClone?: boolean, skipNone?: boolean, typeName?: string }} [options]
 * @returns {Promise<{ ok: true, code: string, warning?: string } | { ok: false, error: string }>}
 *
 * quicktype-core Rust renderer 行为：
 * - `just-types: true`  只输出 struct 声明，不输出 serde_json 反序列化代码
 * - `derive-debug: true` (默认)  为类型 derive Debug
 * - `derive-clone: true` (默认)  为类型 derive Clone
 * - `skip-serializing-none: true`  对 Option 字段加 `#[serde(skip_serializing_none)]`
 *
 * `typeName` 作为 `addSource({ name })` 传入，控制顶层 struct 名。非合法 Rust 标识符时
 * 回退到 'AutoGenerated' 并返回 warning，让 UI 提示用户。
 */
const RS_IDENT_RE = /^[A-Za-z_][A-Za-z0-9_]*$/

export async function jsonToRust(jsonString, {
  deriveDebug = true,
  deriveClone = true,
  skipNone = false,
  typeName = 'AutoGenerated',
} = {}) {
  if (!jsonString || jsonString.trim() === '') {
    return { ok: true, code: '' }
  }

  let parsed
  try {
    parsed = JSON.parse(jsonString)
  } catch (e) {
    return { ok: false, error: 'JSON 解析失败：' + e.message }
  }

  const trimmedName = (typeName || '').trim()
  const isValidName = trimmedName === '' || RS_IDENT_RE.test(trimmedName)
  const safeName = isValidName ? (trimmedName || 'AutoGenerated') : 'AutoGenerated'
  const warning = isValidName
    ? undefined
    : `结构体名 "${trimmedName}" 不是合法 Rust 标识符`

  try {
    const jsonInput = jsonInputForTargetLanguage('rust')
    await jsonInput.addSource({ name: safeName, samples: [JSON.stringify(parsed)] })
    const inputData = new InputData()
    inputData.addInput(jsonInput)

    const rendererOptions = {
      'just-types': true,
      'derive-debug': deriveDebug,
      'derive-clone': deriveClone,
    }
    if (skipNone) rendererOptions['skip-serializing-none'] = true

    const result = await quicktype({
      lang: 'rust',
      inputData,
      rendererOptions,
    })
    return { ok: true, code: result.lines.join('\n'), warning }
  } catch (e) {
    return { ok: false, error: '生成失败：' + e.message }
  }
}
```

```js
  lombok = false,
  typeName = 'AutoGenerated',
} = {}) {
  if (!jsonString || jsonString.trim() === '') {
    return { ok: true, code: '' }
  }

  let parsed
  try {
    parsed = JSON.parse(jsonString)
  } catch (e) {
    return { ok: false, error: 'JSON 解析失败：' + e.message }
  }

  const trimmedName = (typeName || '').trim()
  const isValidName = trimmedName === '' || JAVA_IDENT_RE.test(trimmedName)

    ? undefined

  try {
    const jsonInput = jsonInputForTargetLanguage('java')
    await jsonInput.addSource({ name: safeName, samples: [JSON.stringify(parsed)] })
    const inputData = new InputData()
    inputData.addInput(jsonInput)

    const rendererOptions = {
      'just-types': true,
      'array-type': useList ? 'list' : 'array',
    }
    if (lombok) rendererOptions['lombok'] = true

    const result = await quicktype({
      lang: 'java',
      inputData,
      rendererOptions,
    })
    return { ok: true, code: result.lines.join('\n'), warning }
  } catch (e) {

  }
```

```js
import { quicktype, jsonInputForTargetLanguage, InputData } from 'quicktype-core'

/**
 * @param {string} jsonString
 * @param {{ namespace?: string, typeName?: string }} [options]
 * @returns {Promise<{ ok: true, code: string, warning?: string } | { ok: false, error: string }>}
 *
 * quicktype-core C++ renderer 行为：
 * - `just-types: true`  只输出 struct 声明，不输出 nlohmann json 反序列化代码
 * - `namespace`  生成的命名空间名，默认 'quicktype'
 *
 * `typeName` 作为 `addSource({ name })` 传入，控制顶层 struct 名。非合法 C++ 标识符时
 * 回退到 'AutoGenerated' 并返回 warning，让 UI 提示用户。
 */
const CPP_IDENT_RE = /^[A-Za-z_][A-Za-z0-9_]*$/

export async function jsonToCpp(jsonString, {
  namespace = 'quicktype',
  typeName = 'AutoGenerated',
} = {}) {
  if (!jsonString || jsonString.trim() === '') {
    return { ok: true, code: '' }
  }

  let parsed
  try {
    parsed = JSON.parse(jsonString)
  } catch (e) {
    return { ok: false, error: 'JSON 解析失败：' + e.message }
  }

  const trimmedName = (typeName || '').trim()
  const isValidName = trimmedName === '' || CPP_IDENT_RE.test(trimmedName)
  const safeName = isValidName ? (trimmedName || 'AutoGenerated') : 'AutoGenerated'
  const warning = isValidName
    ? undefined
    : `结构体名 "${trimmedName}" 不是合法 C++ 标识符`

  try {
    const jsonInput = jsonInputForTargetLanguage('cpp')
    await jsonInput.addSource({ name: safeName, samples: [JSON.stringify(parsed)] })
    const inputData = new InputData()
    inputData.addInput(jsonInput)

    const rendererOptions = {
      'just-types': true,
      namespace,
    }

    const result = await quicktype({
      lang: 'cpp',
      inputData,
      rendererOptions,
    })
    return { ok: true, code: result.lines.join('\n'), warning }
  } catch (e) {
    return { ok: false, error: '生成失败：' + e.message }
  }
}
```

```js
  }

  const trimmedName = (typeName || '').trim()
  const isValidName = trimmedName === '' || CSHARP_IDENT_RE.test(trimmedName)
  const safeName = isValidName ? (trimmedName || 'AutoGenerated') : 'AutoGenerated'
  const warning = isValidName
    ? undefined
    : `结构体名 "${trimmedName}" 不是合法 C# 标识符`

  try {
    const jsonInput = jsonInputForTargetLanguage('csharp')
    await jsonInput.addSource({ name: safeName, samples: [JSON.stringify(parsed)] })
    const inputData = new InputData()
    inputData.addInput(jsonInput)

    const rendererOptions = {
      'just-types': true,
      namespace,
      'array-type': useList ? 'list' : 'array',
    }

    const result = await quicktype({
      lang: 'csharp',
      inputData,
      rendererOptions,
    })
    return { ok: true, code: result.lines.join('\n'), warning }
  } catch (e) {
    return { ok: false, error: '生成失败：' + e.message }
  }
}
```
